int test_function(int a, int b) {
    int result;
    bool condition;
    
    condition = (a > b);
    
    if (condition) {
        result = a;
    } else {
        result = b;
    }
    
    return result;
}

int main() {
    int num1;

}

int main() {
    int counter;
    int sum;
    int limit;
    
    counter = 0;
    sum = 0;
    limit = 10;
    
    while (counter < limit) {
        sum = sum + counter;
        counter = counter + 1;
        
        if (counter == 5) {
            break;
        }
    }
    
    return sum;
}
